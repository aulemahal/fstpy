   df = df1.query('name.str.contains("gram",na=False)',engine='python')

   df.to_pickle('test_data_1.pkl.bz2')

   df2 = pd.read_pickle('test_data_1.pkl.bz2')

   import numpy as np
   d = [
        { 'nom':'Holly','prenom':'Matthew','classification':'TI','niveau':4},

   df = pd.DataFrame(d)
